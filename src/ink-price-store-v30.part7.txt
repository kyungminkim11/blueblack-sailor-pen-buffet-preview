function syncUrl(){const url=new URL(location.href);if(query.trim())url.searchParams.set('q',query.trim());else url.searchParams.delete('q');url.searchParams.delete('selection');history.replaceState(null,'',url);}
function shareUrl(){const url=new URL(location.href);if(query.trim())url.searchParams.set('q',query.trim());else url.searchParams.delete('q');if(list.length)url.searchParams.set('selection',encodeSelection(persistableList()));else url.searchParams.delete('selection');return url.href;}
async function shareList(){
  if(!list.length)return;
  const lines=list.map((item,index)=>`${index+1}. ${displayName(item)} · ${item.volume} · ${formatWon(item.price)}`);
  const body=`BlueBlack Ink Check List\n${lines.join('\n')}\n\n${text().total}: ${formatWon(listTotal())}\n${text().staffNote}`;
  const url=shareUrl();
  if(navigator.share){try{await navigator.share({title:text().staffTitle,text:body,url});return;}catch{}}
  try{await navigator.clipboard.writeText(`${body}\n${url}`);toast(text().shared);}catch{window.prompt(text().shared,`${body}\n${url}`);}
}
function clearForNewCustomer(){if(list.length&&!confirm(text().clearConfirm))return;list=[];sessionStorage.removeItem(SESSION_KEY);const url=new URL(location.href);url.searchParams.delete('selection');history.replaceState(null,'',url);writeList();document.querySelector('.ink-store-search')?.focus();}

function staffDialog(){
  let dialog=document.querySelector('.ink-store-dialog');if(dialog)return dialog;
  dialog=document.createElement('dialog');dialog.className='ink-store-dialog';dialog.innerHTML='<div class="ink-store-dialog-shell"><div class="ink-store-dialog-head"><div><small>BLUEBLACK INK SERVICE</small><h2></h2></div><button type="button" class="ink-store-dialog-close" aria-label="Close">×</button></div><div class="ink-store-dialog-list"></div><div class="ink-store-dialog-total"><span></span><strong></strong></div><p class="ink-store-dialog-note"></p><div class="ink-store-dialog-actions"><button type="button" class="ink-store-dialog-share primary"></button><button type="button" class="ink-store-dialog-print"></button></div></div>';
  document.body.append(dialog);
  dialog.querySelector('.ink-store-dialog-close').addEventListener('click',()=>dialog.close());
  dialog.querySelector('.ink-store-dialog-share').addEventListener('click',shareList);
  dialog.querySelector('.ink-store-dialog-print').addEventListener('click',()=>window.print());
  dialog.addEventListener('click',event=>{if(event.target===dialog)dialog.close();});
  return dialog;
}
function openStaffDialog(){
  if(!list.length)return;
  const dialog=staffDialog();dialog.querySelector('h2').textContent=text().staffTitle;
  dialog.querySelector('.ink-store-dialog-list').innerHTML=list.map((item,index)=>`<div><span>${index+1}</span><p><b>${escapeHtml(displayName(item))}</b><small>${item.volume} · ${formatWon(item.price)}</small></p></div>`).join('');
  dialog.querySelector('.ink-store-dialog-total span').textContent=text().total;dialog.querySelector('.ink-store-dialog-total strong').textContent=formatWon(listTotal());
  dialog.querySelector('.ink-store-dialog-note').textContent=text().staffNote;dialog.querySelector('.ink-store-dialog-share').textContent=text().share;dialog.querySelector('.ink-store-dialog-print').textContent=text().print;
  if(!dialog.open)dialog.showModal();
}
