function resultStats(values){
  const brands=new Set(values.map(entry=>entry.group.id));
  const series=new Set(values.map(entry=>entry.item.id));
  const colors=new Set(values.filter(entry=>entry.color).map(entry=>entry.color.id));
  return{brands:brands.size,series:series.size,colors:colors.size};
}

function renderResults(){
  const root=document.querySelector('.ink-store-results');if(!root)return;
  const values=filteredEntries();
  const stats=resultStats(values);
  document.querySelector('[data-ink-store-stats]').textContent=query?text().stats(stats.brands,stats.series,stats.colors):text().initialBody;
  const shown=values.slice(0,visibleLimit);
  if(!shown.length){root.innerHTML=`<div class="ink-store-empty"><strong>${text().emptyTitle}</strong><span>${text().emptyBody}</span></div>`;renderMore(0,0);return;}
  root.innerHTML=shown.map(resultCard).join('');
  root.querySelectorAll('.ink-store-volume').forEach(button=>button.addEventListener('click',()=>toggleSelection(button)));
  renderMore(shown.length,values.length);
}
function renderMore(shown,total){

  const remaining=Math.max(0,total-shown);button.hidden=remaining===0;button.innerHTML=remaining?`${text().more}<small>${text().remaining(remaining)}</small>`:'';
}
